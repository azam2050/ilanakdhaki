GROUP BY category

const hashCustomer = (value) => {
  return crypto.createHash('sha256')
    .update(value.toLowerCase().trim())
    .digest('hex')
}

// تشفير توكنات OAuth — AES-256
const encryptToken = (token) => {
  const iv = crypto.randomBytes(16)
  const cipher = crypto.createCipheriv(
    'aes-256-cbc',
    Buffer.from(process.env.ENCRYPTION_KEY),
    iv
  )
  return iv.toString('hex') + ':' + 
    cipher.update(token, 'utf8', 'hex') + 
    cipher.final('hex')
}

const verifyWebhook = (payload, signature) => {
  const expected = crypto
    .createHmac('sha256', process.env.SALLA_WEBHOOK_SECRET)
    .update(payload)
    .digest('hex')
  return crypto.timingSafeEqual(
    Buffer.from(signature),
    Buffer.from(expected)
  )
}

// Idempotency — منع معالجة Webhook مكرر
const processWebhook = async (webhookId, handler) => {
  const key = `webhook:${webhookId}`
  const processed = await redis.get(key)
  if (processed) return  // تم معالجته مسبقاً
  
  await handler()
  await redis.setex(key, 86400, '1')  // 24 ساعة
}

إعداد الدومين الخاص:
# Nginx Reverse Proxy
server {
  listen 443 ssl;
  server_name yourbrand.com;
  
  # إخفاء هوية ريبلت
  proxy_hide_header X-Powered-By;
  proxy_hide_header Server;
  
  location / {
    proxy_pass https://your-replit-app.repl.co;
    proxy_set_header Host yourbrand.com;
    proxy_set_header X-Real-IP $remote_addr;
  }
}

٩.٢ أولويات لا تقبل التفاوض
🔴 هذه النقاط يجب تنفيذها من اليوم الأول:
الأمان: التشفير، Rate Limiting، النسخ الاحتياطي — ليست اختيارية أبداً
الموافقات: شاشة الموافقة القانونية يجب أن تكون واضحة قبل أي سحب للبيانات
عزل التجار: كل استعلام يتحقق من صلاحية التاجر — تاجر لا يرى بيانات آخر أبداً
Idempotency: كل Webhook يُعالج مرة واحدة فقط حتى لو وصل مكرراً
Graceful Degradation: إذا تعطلت Claude API يستمر النظام بقواعد رياضية
إخفاء ريبلت: لا يظهر أي أثر لريبلت في الرابط أو Headers أو الكود المصدري
٩.٣ اختبارات القبول قبل الإطلاق
تثبيت تطبيق سلة وتطبيق زد واستقبال الـ Webhooks بنجاح
ربط حساب سناب وميتا وتيك توك وجوجل وإطلاق حملة تجريبية
استقبال ١,٠٠٠ حدث Webhook في الدقيقة دون تأخير
قطع الاتصال بـ Claude API والتحقق أن النظام يستمر
محاولة الوصول لبيانات تاجر من حساب تاجر آخر والتحقق من الرفض
اختبار استعادة قاعدة البيانات من نسخة احتياطية في أقل من ٣٠ دقيقة
التحقق أن الدومين لا يكشف أي headers تتعلق بريبلت
